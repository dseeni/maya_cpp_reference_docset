106  def draw(self, view, path, style, status):
107  ## Get the size
108  ##
109  thisNode = self.thisMObject()
110  plug = om.MPlug( thisNode, footPrint.size )
111  sizeVal = plug.asMDistance()
112  multiplier = sizeVal.asCentimeters()
113 
114  global sole, soleCount
115  global heel, heelCount
116 
117  view.beginGL()
118 
119  ## drawing in VP1 views will be done using V1 Python APIs:
120  import maya.OpenMayaRender as v1omr
121  glRenderer = v1omr.MHardwareRenderer.theRenderer()
122  glFT = glRenderer.glFunctionTable()
123 
124  if ( style == omui.M3dView.kFlatShaded ) or ( style == omui.M3dView.kGouraudShaded ):
125  ## Push the color settings
126  ##
127  glFT.glPushAttrib( v1omr.MGL_CURRENT_BIT )
128 
129  # Show both faces
130  glFT.glDisable( v1omr.MGL_CULL_FACE )
131 
132  if status == omui.M3dView.kActive:
133  view.setDrawColor( 13, omui.M3dView.kActiveColors )
134  else:
135  view.setDrawColor( 13, omui.M3dView.kDormantColors )
136 
137  glFT.glBegin( v1omr.MGL_TRIANGLE_FAN )
138  for i in xrange(soleCount-1):
139  glFT.glVertex3f( sole[i][0] * multiplier, sole[i][1] * multiplier, sole[i][2] * multiplier )
140  glFT.glEnd()
141 
142  glFT.glBegin( v1omr.MGL_TRIANGLE_FAN )
143  for i in xrange(heelCount-1):
144  glFT.glVertex3f( heel[i][0] * multiplier, heel[i][1] * multiplier, heel[i][2] * multiplier )
145  glFT.glEnd()
146 
147  glFT.glPopAttrib()
148 
149  ## Draw the outline of the foot
150  ##
151  glFT.glBegin( v1omr.MGL_LINES )
152  for i in xrange(soleCount-1):
153  glFT.glVertex3f( sole[i][0] * multiplier, sole[i][1] * multiplier, sole[i][2] * multiplier )
154  glFT.glVertex3f( sole[i+1][0] * multiplier, sole[i+1][1] * multiplier, sole[i+1][2] * multiplier )
155 
156  for i in xrange(heelCount-1):
157  glFT.glVertex3f( heel[i][0] * multiplier, heel[i][1] * multiplier, heel[i][2] * multiplier )
158  glFT.glVertex3f( heel[i+1][0] * multiplier, heel[i+1][1] * multiplier, heel[i+1][2] * multiplier )
159  glFT.glEnd()

161  view.endGL()

243  def updateRenderItems(self, dagPath, renderList ):
244 
245  fullItemList = (
246  # Wireframe sole and heel:
247  ( (footPrintGeometryOverride.wireframeHeelItemName_,
248  footPrintGeometryOverride.wireframeSoleItemName_),
249  omr.MGeometry.kLineStrip,
250  omr.MGeometry.kWireframe),
251 
252  # Shaded sole and heel
253  ( (footPrintGeometryOverride.shadedHeelItemName_,
254  footPrintGeometryOverride.shadedSoleItemName_),
255  omr.MGeometry.kTriangleStrip,
256  omr.MGeometry.kShaded)
257  )
258 
259  for itemNameList, geometryType, drawMode in fullItemList:
260  for itemName in itemNameList:
261  renderItem = None
262  index = renderList.indexOf(itemName)
263  if index < 0:
264  renderItem = omr.MRenderItem.create(
265  itemName,
266  omr.MRenderItem.DecorationItem,
267  geometryType)
268  renderItem.setDrawMode(drawMode)
269  renderItem.setDepthPriority(5)
270 
271  renderList.append(renderItem)
272  else:
273  renderItem = renderList[index]
274 
275  if renderItem:
276  if self.mSolidUIShader:
277  self.mSolidUIShader.setParameter(footPrintGeometryOverride.colorParameterName_, omr.MGeometryUtilities.wireframeColor(dagPath))
278  renderItem.setShader(self.mSolidUIShader)
279  renderItem.enable(True)

281  def populateGeometry(self, requirements, renderItems, data):
282  vertexBufferDescriptorList = requirements.vertexRequirements()
283 
284  for vertexBufferDescriptor in vertexBufferDescriptorList:
285  if vertexBufferDescriptor.semantic == omr.MGeometry.kPosition:
286  verticesCount = soleCount+heelCount
287  verticesBuffer = data.createVertexBuffer(vertexBufferDescriptor)
288  verticesPositionDataAddress = verticesBuffer.acquire(verticesCount, True)
289  verticesPositionData = ((ctypes.c_float * 3)*verticesCount).from_address(verticesPositionDataAddress)
290 
291  verticesPointerOffset = 0
292 
293  # We concatenate the heel and sole positions into a single vertex buffer.
294  # The index buffers will decide which positions will be selected for each render items.
295  for vtxList in (heel, sole):
296  for vtx in vtxList:
297  verticesPositionData[verticesPointerOffset][0] = vtx[0] * self.mMultiplier
298  verticesPositionData[verticesPointerOffset][1] = vtx[1] * self.mMultiplier
299  verticesPositionData[verticesPointerOffset][2] = vtx[2] * self.mMultiplier
300  verticesPointerOffset += 1
301 
302  verticesBuffer.commit(verticesPositionDataAddress)
303 
304  break
305 
306  for item in renderItems:
307  if not item:
308  continue
309 
310  # Start position in the index buffer (start of heel or sole positions):
311  startIndex = 0
312  endIndex = 0
313  # Number of index to generate (for line strip, or triangle list)
314  numIndex = 0
315  isWireFrame = True
316 
317  if item.name() == footPrintGeometryOverride.wireframeHeelItemName_:
318  numIndex = heelCount
319  elif item.name() == footPrintGeometryOverride.wireframeSoleItemName_:
320  startIndex = heelCount
321  numIndex = soleCount
322  elif item.name() == footPrintGeometryOverride.shadedHeelItemName_:
323  numIndex = heelCount - 2
324  startIndex = 1
325  endIndex = heelCount - 2
326  isWireFrame = False
327  elif item.name() == footPrintGeometryOverride.shadedSoleItemName_:
328  startIndex = heelCount
329  endIndex = heelCount + soleCount - 2
330  numIndex = soleCount - 2
331  isWireFrame = False
332 
333  if numIndex:
334  indexBuffer = data.createIndexBuffer(omr.MGeometry.kUnsignedInt32)
335  indicesAddress = indexBuffer.acquire(numIndex, True)
336  indices = (ctypes.c_uint * numIndex).from_address(indicesAddress)
337 
338  i = 0
339  while i < numIndex:
340  if isWireFrame:
341  # Line strip starts at first position and iterates thru all vertices:
342  indices[i] = startIndex + i
343  i += 1
344  else:
345  # Triangle strip
346  indices[i] = startIndex + i/2
347  if i+1 < numIndex:
348  indices[i+1] = endIndex - i/2
349  i += 2
350 
351  indexBuffer.commit(indicesAddress)
352  item.associateWithIndexBuffer(indexBuffer)
353 
354  mMultiplierChanged = False
